    const SOUND_SOURCES = {

        click: 'https://files.catbox.moe/998q29.mp3',

    const SOUNDS = {};

    let loadedCount = 0;

    const total = Object.keys(SOUND_SOURCES).length;


    Object.entries(SOUND_SOURCES).forEach(([key, src]) => {

        const audio = new Audio(src);

        audio.addEventListener('canplaythrough', () => {

            if (!SOUNDS[key]) {

                SOUNDS[key] = audio;

                loadedCount++;

                if (loadedCount === total) {

                    document.getElementById('loading-toast').style.display = 'none';

                }

            }

        }, { once: true });

        audio.load();

    });


    function playSnd(name) {

        if (SOUNDS[name]) {

            SOUNDS[name].currentTime = 0;

            SOUNDS[name].play().catch(() => {});

        }

    }

    const VOWELS_MAP = { 'a': '1', 'e': '2', 'o': '3', 'i': '4', 'y': '5', 'u': '5' };

    const REV_VOWELS = { '1': 'a', '2': 'e', '3': 'o', '4': 'i', '5': 'u' };

    const TRANSLIT = {'а':'a','б':'b','в':'v','г':'g','д':'d','е':'e','ё':'e','ж':'zh','з':'z','и':'i','й':'i','к':'k','л':'l','м':'m','н':'n','о':'o','п':'p','р':'r','с':'s','т':'t','у':'u','ф':'f','х':'h','ц':'ts','ч':'ch','ш':'sh','щ':'sch','ы':'i','э':'e','ю':'yu','я':'ya'};

    const REV_TRANSLIT = {'a':'а','b':'б','v':'в','g':'г','д':'д','e':'е','zh':'ж','z':'з','i':'и','k':'к','l':'л','m':'м','n':'н','o':'о','p':'п','r':'р','с':'с','т':'т','u':'у','f':'ф','h':'х','ts':'ц','ch':'ч','sh':'ш','sch':'щ','yu':'ю','ya':'я'};


    function process(isEnc) {

        playSnd('click');

        const text = document.getElementById('input').value.toLowerCase();

        if(!text) return;

        let words = text.split(/\s+/), res = [], ru = [];

        words.forEach(word => {

            if (word.length < 2) { res.push(word); ru.push(word); return; }

            if (isEnc) {

                let t = word.split('').map(c => TRANSLIT[c] || c).join('');

                let a = t.split('');

                [a[0], a[a.length-1]] = [a[a.length-1], a[0]];

                res.push(a.map(c => VOWELS_MAP[c] || c).join(''));

            } else {

                let a = word.split('').map(c => REV_VOWELS[c] || c);

                [a[0], a[a.length-1]] = [a[a.length-1], a[0]];

                let lat = a.join('');

                res.push(lat);

                let rW = "";

                for(let i=0; i<lat.length; i++) {

                    let d = lat[i]+lat[i+1];

                    if(REV_TRANSLIT[d]){ rW+=REV_TRANSLIT[d]; i++; }

                    else { rW+=REV_TRANSLIT[lat[i]]||lat[i]; }

                }

                ru.push(rW);

            }

        });

        document.getElementById('output').value = res.join(' ');

        document.getElementById('ru-hint-box').innerText = isEnc ? "" : "RU: " + ru.join(' ');

    }

    }
